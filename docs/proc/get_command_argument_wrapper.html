  subroutine get_command_argument_wrapper(arg)
    character(*), intent(out):: arg
    integer :: ii

    ii = 1
    do
      call get_command_argument(ii, arg)

      if (len_trim(arg) == 0) then
        arg = trim(arg)
      else
        exit
      end if

      ii = ii + 1
    end do

    return
  end subroutine get_command_argument_wrapper
